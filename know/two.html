# 第二章 webgl简单入门教程

# webgl在画布刷底色

    1. 在html中建立canvas 画布

    <canvas id="canvas"></canvas>

    2. 在js中获取canvas 画布

    const canvas = document.getElementById('canvas')

    3. 使用canvas 获取webgl 绘图上下文

    const gl = canvas.getContext('webgl')

    4. 指定将要用来清空绘图区的颜色

    gl.clearColor(0,0,0,1)

    5. 使用之前指定的颜色，清空绘图区

    gl.clear(gl_COLOR_BUFFER_BIT)

    完整代码

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
  <style>
    * {
      margin: 0;
      padding: 0;
    }
    #canvas {
      background-color: #000;
    }
  </style>
</head>
<body>
  <!-- 在html中建立canvas 画布 -->
  <canvas id="canvas"></canvas>
  <script>
    // 在js中获取canvas 画布
    const canvas = document.querySelector('#canvas')
    // 设置canvas画布宽高
    canvas.width = window.innerWidth
    canvas.height = window.innerHeight
    // 获取webgl绘图上下文
    const gl = canvas.getContext('webgl')
    // 指定将要用来清空绘图区的颜色
    gl.clearColor(1,1,1,1)
    // 使用之前指定的颜色，清空绘图区
    gl.clear(gl.COLOR_BUFFER_BIT)
  </script>
</body>
</html>

# css颜色转webgl颜色

    1. 定义css颜色

    const rgbaCss = '(0,255,255,1)'

     2. 定义正则      \(\)表示先找到小括号 (.*)的()表示捕捉括号,(.*)捕捉括号里的所有东西

    const reg = /\((.*)\)/

     3. 捕捉数据      reg.exec(rgbaCss) 结果是个数组["(0,255,255,1)","0,255,255,1"]

     const rgbaStr = reg.exec(rgbaCss)[1]

     4. 加工数据      webgl颜色r,g,b,a范围是0-1

     const [r,g,b,a] = glColor.split(',').map((n,i) =>i<3?+n/255:+n)

     5. 声明颜色

     gl.clearColor(r,g,b,a)

     6. 刷底色

     gl.clear(gl.COLOR_BUFFER_BIT)

     完整代码

<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
  <style>
    * {
      margin: 0;
      padding: 0;
    }

    #canvas {
      background-color: #000;
    }
  </style>
</head>

<body>
  <!-- 在html中建立canvas 画布 -->
  <canvas id="canvas"></canvas>
  <script>
    // 在js中获取canvas 画布
    const canvas = document.querySelector('#canvas')
    // 设置canvas画布宽高
    canvas.width = window.innerWidth
    canvas.height = window.innerHeight
    // 获取webgl绘图上下文
    const gl = canvas.getContext('webgl')
    // 指定将要用来清空绘图区的颜色
    gl.clearColor(1, 1, 1, 1)
    // 使用之前指定的颜色，清空绘图区
    gl.clear(gl.COLOR_BUFFER_BIT)
    const rgbaCss = '(0,255,255,1)'
    const reg = /\((.*)\)/
    const rgbaStr = reg.exec(rgbaCss)[1]
    const [r, g, b, a] = rgbaStr.split(',').map((n, i) => i < 3 ? +n / 255 : +n)

    // 声明颜色
    gl.clearColor(r, g, b, a)
    // 刷底色
    gl.clear(gl.COLOR_BUFFER_BIT)
  </script>
</body>

</html>

# 案列---多姿多彩的画布

     用到threejs的Color库

     链接地址: http://www.yanhuangxueyuan.com/threejs/docs/index.html#api/zh/math/Color

     完整代码

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>多彩画布</title>
  <style>
    body {
      margin:0;
      overflow: hidden;
    }
    #canvas {
      background-color: pink;
    }
  </style>
</head>
<body>
  <canvas id="canvas"></canvas>
  <script type="module">
    // 引入threejs Color对象
    import { Color } from 'https://unpkg.com/three/build/three.module.js'
    const canvas = document.querySelector('#canvas')
    canvas.width = window.innerWidth
    canvas.height = window.innerHeight
    // 3维画笔
    const gl = canvas.getContext('webgl')
    gl.clearColor(1,0,0,1)
    gl.clear(gl.COLOR_BUFFER_BIT)

    // 实例化Color对象
    const color = new Color('rgba(255,0,255,1)')
    const {r,g,b} = color
    gl.clearColor(r,g,b,1)
    gl.clear(gl.COLOR_BUFFER_BIT)

    // 动画驱动
    !(function ani() {
      // 色相偏移 色相(H) 饱和度(S) 亮度(L)
      color.offsetHSL(0.005,0,0)
      const {r,g,b} = color
      gl.clearColor(r,g,b,1)
      gl.clear(gl.COLOR_BUFFER_BIT)
      requestAnimationFrame(ani)
    })()
  </script>
</body>
</html>

# canvas 2d坐标系

     canvas 2d 坐标系的原点在左上角。

     canvas 2d 坐标系的y 轴方向是朝下的。

     canvas 2d 坐标系的坐标基底有两个分量，分别是一个像素的宽和一个像素的高，即1个单位的宽便是1个像素的宽，1个单位的高便是一个像素的高。

    

# webgl坐标系

     webgl坐标系的坐标原点在画布中心。

     webgl坐标系的y 轴方向是朝上的。

     webgl坐标基底中的两个分量分别是半个canvas的宽和canvas的高，即1个单位的宽便是半个个canvas的宽，1个单位的高便是半个canvas的高。
